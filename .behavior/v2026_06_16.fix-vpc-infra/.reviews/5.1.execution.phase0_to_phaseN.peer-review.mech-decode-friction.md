  GatewayId?: string;
  NatGatewayId?: string;
}> => {
  if (route.target.gatewayInternet) {
    const igwRef = route.target.gatewayInternet;
    if ('id' in igwRef) return { GatewayId: igwRef.id };
    if ('exid' in igwRef) {
      const igw = await getOneVpcInternetGateway(
        { by: { unique: igwRef } },
        context,
      );
      if (!igw)
        throw new UnexpectedCodePathError(
          'internet gateway not found for route',
          { igwRef },
        );
      return { GatewayId: igw.id };
    }
    throw new UnexpectedCodePathError('invalid internet gateway ref', {
      igwRef,
    });
  }
  if (route.target.gatewayNat) {
    return { NatGatewayId: route.target.gatewayNat.id };
  }
  throw new UnexpectedCodePathError('route lacks target', { route });
})();
