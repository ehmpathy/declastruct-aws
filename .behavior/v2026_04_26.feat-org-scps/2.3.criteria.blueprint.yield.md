## subcomponent contracts

### DeclaredAwsOrganizationServiceControlPolicy domain object

given('DeclaredAwsOrganizationServiceControlPolicy contract')
  then('extends DomainEntity')
  then('has primary key: id')
  then('has unique key: name')
  then('has required field: content (DeclaredAwsIamPolicyDocument)')
  then('has optional field: description (string)')
  then('has optional field: tags (DeclaredAwsTags)')
  then('has metadata fields: id, arn (assigned by AWS)')

### DeclaredAwsOrganizationServiceControlPolicyAttachment domain object

given('DeclaredAwsOrganizationServiceControlPolicyAttachment contract')
  then('extends DomainLiteral')
    sothat('it has no lifecycle — attachment is a relationship, not an entity')
  then('has required field: policy (RefByUnique)')
  then('has required field: target (RefByPrimary of Organization or OrganizationAccount)')

### daoAwsOrganizationServiceControlPolicy

given('daoAwsOrganizationServiceControlPolicy.get contract')
  then('exposes: get.one.byUnique({ name }) => SCP | null')
  then('exposes: get.one.byPrimary({ id }) => SCP | null')
  then('throws ConstraintError when duplicate names detected')

given('daoAwsOrganizationServiceControlPolicy.set contract')
  then('exposes: set.findsert(scp) => SCP with id and arn')
  then('is idempotent: returns extant if name matches')

given('daoAwsOrganizationServiceControlPolicy.del contract')
  then('exposes: del.one.byUnique({ name }) => void')
  then('exposes: del.one.byPrimary({ id }) => void')
  then('throws ConstraintError when SCP is still attached')

### daoAwsOrganizationServiceControlPolicyAttachment

given('daoAwsOrganizationServiceControlPolicyAttachment.get contract')
  then('exposes: get.all.forPolicy({ policy }) => Attachment[]')
  then('exposes: get.all.forTarget({ target }) => Attachment[]')

given('daoAwsOrganizationServiceControlPolicyAttachment.set contract')
  then('exposes: set.findsert(attachment) => Attachment')
  then('is idempotent: no-op if already attached')
  then('throws ConstraintError when policy not found')
  then('throws ConstraintError when target not found')

given('daoAwsOrganizationServiceControlPolicyAttachment.del contract')
  then('exposes: del.one({ policy, target }) => void')
  then('throws ConstraintError when detach would leave target with zero SCPs')
